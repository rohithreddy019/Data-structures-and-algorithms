#include <cmath>
#include <cstdio>
#include <vector>
#include <iostream>
#include <algorithm>
using namespace std;


int main()
{
    int x;
    cin>>x;
    for(int i=1;i<=x;i++)
        {
            long int y,u;
            cin>>y;
            u=y;
            int c=0;
            while(y)
            {
                y=y>>1;
                c++;
            }
            int k1=c/2;
            int k2=c-k1-1;
            long long int k3=1<<k1;
            long long int k4=1<<k2;
            long long int k=k3*k4;
            if(k==u)
                cout<<"True"<<endl;
            else
                cout<<"False"<<endl;
      
        }
    return 0;
}
